// Add new migration with name
dotnet ef migrations add InitialMigration -c InventoryDbContext -o Database/Migrations

// Remove last migration. Note that you must be on old migration before remove the last one
dotnet ef migrations remove -c InventoryDbContext

// Database update
dotnet ef database update -c InventoryDbContext

// Database update to specific migration
dotnet ef database update MigrationSpecificName -c InventoryDbContext
